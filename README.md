## Install

```bash
npm install --save react-datasheet-grid
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-datasheet-grid'
import 'react-datasheet-grid/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
